def create_standup_file(file, fixture = 'previous_entry')
  dir = File.dirname(file)
  FileUtils.mkdir(dir) unless File.directory?(dir)
  File.open(file, 'w') do |f|
    f.puts fixtures[fixture]
    f.puts
    f.puts fixtures['current_entry']
  end
end

    @return [Boolean] true if successful

def reset
  @date = Date.today
  @editor = set_editor
  @verbose = false
  @edit = true
  @write = true
  @print = false
  @auto_fill_previous = true
  @preference_file = ::File.expand_path(::File.join(ENV['HOME'], '.standuprc'))
end

def setup
  @workdir = File.join(__dir__, 'files')
  @current_test_file =
    File.join(@workdir, Date.today.strftime('%Y_%m.md'))
  @previous_month_test_file =
    File.join(@workdir, Date.today.prev_month.strftime('%Y_%m.md'))
end

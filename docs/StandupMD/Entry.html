  new(
    Date.today,
    config.current,
    config.previous,
    config.impediments,
    config.notes
  ).tap { |entry| yield entry if block_given? }
end

def initialize(date, current, previous, impediments, notes = [])
  raise unless date.is_a?(Date)
  @config = self.class.config

  @date = date
  @current = current
  @previous = previous
  @impediments = impediments
  @notes = notes
end

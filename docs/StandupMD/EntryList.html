def filter(start_date, end_date)
  self.class.new(
    *@entries.select { |e| e.date.between?(start_date, end_date) }
  )
end

def to_h
  @entries.map do |e|
    [
      e.date,
      {
        "current" => e.current,
        "previous" => e.previous,
        "impediments" => e.impediments,
        "notes" => e.notes
      }
    ]
  end.to_h
end
